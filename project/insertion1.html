for (c = 0; c < n; c++)
scanf("%d", &array[c]);
printf("Please enter the location where you want to insert an new element\n");
scanf("%d", &position);

scanf("%d", &value);
for (c = n - 1; c >= position - 1; c--)
array[c+1] = array[c];
array[position-1] = value;

for (c = 0; c <= n; c++)
