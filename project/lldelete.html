Delete Elements in Linked Lists

Source Code:

typedef struct _node {
int value;
struct node *next;
}node;
int main(int argc, char *argv[])
{
int i;
node *n[10];
node *tmp, *del;
/* Construct a linked list of 10 nodes */
for (i = 0; i < 10; i++)
{
n[i] = (node *) malloc (sizeof(node));
n[i]->value = i + 1;
n[i]->next = NULL;
if (i > 0) {
n[i -1]->next = n[i];
}
}
tmp = n[0];
printf ("Linked List\n");
while (tmp)
{
printf ("Node %p Value %d\n", tmp, tmp->value);
tmp = (node *)tmp->next;
}
printf ("Enter a node to delete :");
scanf ("%p", &tmp);
if (tmp && tmp->next) {
printf ("Deleting %p, value %d\n", tmp, tmp->value);
del = (node *)tmp->next;
*tmp = *del;
free(del);
}
else {
printf ("Either last node or value is NULL, not deleted!\n");
}
printf ("Linked List\n");
tmp = n[0];
while(tmp)
{
printf ("Node %p Value %d\n", tmp, tmp->value);
tmp = tmp->next;
}
return 0;
}
